    import os
    import tempfile
    import pytest

    # Test loading a module from a .py file
    def test_load_module_from_py_file():

        module = load_module_from_file_location(temp_file_path)

        os.remove(temp_file_path)

            temp_file_path = temp_file.name

        os.environ["TEST_ENV_VAR"] = temp_file_path
        module = load_module_from_file_location("${TEST_ENV_VAR}")
        assert module.x == 42
        os.remove(temp_file_path)
